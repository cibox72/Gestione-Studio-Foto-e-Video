function ciboz_generaPDF(p, tipo = 'preventivo') { const { jsPDF } = window.jspdf; const doc = new jsPDF(); const cliente = ciboz_db.clienti[p.clienteIdx] || { cognome: "N/A", nome: "N/A", indirizzo: "" }; // Titolo dinamico: Preventivo o Contratto const titoloDoc = tipo === 'contratto' ? "G&L STUDIO - CONTRATTO DI SERVIZIO" : "G&L STUDIO - PREVENTIVO DI SERVIZIO"; doc.setFontSize(16); doc.setFont(undefined, 'bold'); doc.text(titoloDoc, 105, 15, { align: "center" }); doc.setFontSize(9); doc.setFont(undefined, 'normal'); doc.text("Via Ciarrocco 21, Canicattì (AG) | Tel: 3206337370 - 3669381103 | Email: weddinggel@aruba.it", 105, 22, { align: "center" }); doc.line(20, 26, 190, 26); doc.setFontSize(10); doc.setFont(undefined, 'bold'); doc.text(`Cliente: ${cliente.cognome} ${cliente.nome}`, 20, 35); doc.text(`Indirizzo: ${cliente.indirizzo}`, 20, 40); doc.text(`Data: ${new Date(p.dataCreazione).toLocaleDateString()}`, 140, 35); doc.text(`Evento: ${p.tipo}`, 140, 40); let y = 48; doc.setFontSize(9); const red = [180, 50, 50], black = [0, 0, 0]; function addRow(yPos, label, value, label2, value2) { doc.setTextColor(...red); doc.setFont(undefined, 'bold'); doc.text(label, 20, yPos); doc.setTextColor(...black); doc.setFont(undefined, 'normal'); doc.text(String(value), 50, yPos); if (label2) { doc.setTextColor(...red); doc.setFont(undefined, 'bold'); doc.text(label2, 110, yPos); doc.setTextColor(...black); doc.setFont(undefined, 'normal'); doc.text(String(value2), 145, yPos); } } addRow(y, "Servizio:", p.servizio, "Misura:", p.misura); y += 6; addRow(y, "Orient.:", p.orientamento, "Qta Foto:", p.qta); y += 6; addRow(y, "Copert.:", p.copertina, "Effetti:", p.effetti); y += 6; addRow(y, "Mini Alb.:", p.miniAlbum, "Ingrandim.:", `${p.ingrandimento} (q:${p.ingrandimentoQta})`); y += 6; addRow(y, "Video:", p.video, "Drone:", p.drone); y += 6; if (p.chiesa) { addRow(y, "Chiesa:", p.chiesa); y += 6; } if (p.location) { addRow(y, "Location:", p.location); y += 6; } addRow(y, "Antepr.:", p.anteprima, "Post-Serv.:", p.postServizio); y += 6; doc.setFontSize(8); addRow(y, "Pranzo/Cena STAFF:", p.pranzo); doc.setFontSize(9); y += 6; if (p.note) { doc.setTextColor(...black); doc.setFont(undefined, 'italic'); doc.text(`Note: ${p.note}`, 20, y); y += 6; doc.setFont(undefined, 'normal'); } y += 3; doc.setFontSize(10); doc.setFont(undefined, 'bold'); doc.setTextColor(...red); doc.text("ACCONTI VERSATI:", 20, y); y += 5; if (p.acconti && p.acconti.length > 0) { doc.setFontSize(8); doc.setTextColor(...black); doc.setFont(undefined, 'bold'); doc.text("Tipo", 20, y); doc.text("Data", 60, y); doc.text("Importo", 95, y); doc.text("Pagam.", 130, y); y += 2; doc.line(20, y, 170, y); y += 4; doc.setFont(undefined, 'normal'); p.acconti.forEach(acc => { doc.text(acc.tipo, 20, y); doc.text(acc.data ? new Date(acc.data).toLocaleDateString('it-IT') : '-', 60, y); doc.text('€ ' + (parseFloat(acc.importo) || 0).toFixed(2).replace('.', ','), 95, y); doc.text(acc.metodo, 130, y); y += 5; }); } else { doc.setFontSize(9); doc.setTextColor(...black); doc.text("Nessun acconto versato", 20, y); y += 5; } y += 3; const totale = parseFloat(p.importo) || 0; const totAcc = p.totaleAcconti || 0; const saldo = p.saldo !== undefined ? p.saldo : (totale - totAcc); doc.setDrawColor(0); doc.setFillColor(240, 240, 240); doc.rect(20, y - 3, 170, 18, 'F'); doc.setFontSize(10); doc.setFont(undefined, 'normal'); doc.setTextColor(...black); doc.text(`Totale Servizio:`, 25, y + 3); doc.text(`€ ${totale.toFixed(2).replace('.', ',')}`, 170, y + 3, { align: 'right' }); doc.text(`Totale Acconti:`, 25, y + 9); doc.text(`€ ${totAcc.toFixed(2).replace('.', ',')}`, 170, y + 9, { align: 'right' }); doc.setFont(undefined, 'bold'); doc.setTextColor(39, 174, 96); doc.text(`A SALDO:`, 25, y + 14); doc.text(`€ ${saldo.toFixed(2).replace('.', ',')}`, 170, y + 14, { align: 'right' }); y += 25; // Titolo accettazione dinamico const titoloAccettazione = tipo === 'contratto' ? "ACCETTAZIONE DEL CONTRATTO" : "ACCETTAZIONE DEL PREVENTIVO"; doc.setFontSize(10); doc.setFont(undefined, 'bold'); doc.setTextColor(...black); doc.text(titoloAccettazione, 105, y, { align: "center" }); y += 6; doc.setFontSize(8); doc.setFont(undefined, 'normal'); doc.text("Per accettazione e autorizzazione trattamento dati personali,", 20, y); y += 4; doc.text("si appone firma in calce.", 20, y); y += 12; doc.setLineWidth(0.3); doc.rect(20, y, 75, 25); doc.setFontSize(9); doc.text("Firma Cliente A", 57.5, y + 5, { align: "center" }); doc.rect(115, y, 75, 25); doc.text("Firma Cliente B", 152.5, y + 5, { align: "center" }); y += 32; doc.text("Luogo e Data: ___________________", 20, y); // Nome file dinamico const nomeFile = tipo === 'contratto' ? `Contratto_${cliente.cognome}_${Date.now()}.pdf` : `Preventivo_${cliente.cognome}_${Date.now()}.pdf`; doc.save(nomeFile); }
